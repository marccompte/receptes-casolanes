[Titol]
Amb tòfona

[Ingredients]
persones = 1
pa = Unes llesques de pa de baguette tallades fines
tòfona = Una o dues tòfones negres (poden ser de conserva)
sal = Un pols de sal
oli = Un raig d’oli d’oliva

[Elaboracio]
Tallem les llesques de pa i les torrem.
Laminem la tòfona i la posem a sobre de les llesques de pa.
Amanim amb oli i sal i ja podem començar a gaudir d’aquest bon plat.

[Categories]
categoria = Aperos
